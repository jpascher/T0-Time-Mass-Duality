	\section{Primäre / geometrische Herleitung (Hauptansatz der Theorie)}
	
	\textbf{$D_f = 3 - \xi$} \\
	\begin{itemize}
		\item $\xi = \frac{4}{3} \times 10^{-4}$ (fundamentaler geometrischer Parameter aus Packing / Überlapp von T0-Zellen)
		\item $D_f \approx 2.9998667$ (fraktale Spektraldimension im UV-Limit)
		\item Fraktaler Korrekturfaktor: $K_\text{fraktal} = 1 - 100\xi \approx 0.9866667$
	\end{itemize}
	
	\textbf{Warum genau $3 - \xi$?}
	\begin{itemize}
		\item Eindeutigkeitsbedingung: Nur diese Form lässt alle Kopplungen ($\alpha$, $G$, $m_e/m_\mu$, …) aus \textbf{einem} Parameter $\xi$ emergieren, ohne zusätzliche freie Konstanten.
		\item Geometrisch: 3D-Raum minus minimale fraktale Defizienz durch fundamentale Zell-Überlapp.
	\end{itemize}
	
	\textbf{Quellen:}
	\begin{itemize}
		\item \texttt{xi begründung QFT analyse.md} (Abschnitt „Geometrische Deduktion von $D_f$“)
		\item \texttt{FFGFT\_Narrative\_Master\_De.pdf} (Kapitel „Fraktale Raumstruktur und $\xi$“)

	$\alpha \approx 1/137.035999$ (sehr nah am experimentellen Wert) \\

	\section{Alternative / fitting-basierte Herleitungen (indirekt, empirisch)}
	
	\textbf{Effektiver Faktor $\approx 0.94$ ($D_f \approx 2.94$)}
	\begin{itemize}
		\item Entsteht, wenn man \textbf{nicht} von $D_f = 3 - \xi$ ausgeht, sondern $\xi$ (oder $D_f$) rückwärts aus experimentellen Daten fittet.
		\item Typische Datensätze: Neutrino-Oszillationen, Mixing-Winkel, CMB-Anisotropien, QFT-Loop-Korrekturen, Bell-Test-Grenzen.
		\item Ergebnis: effektiver Suppression-Faktor $\approx 0.94 \dots 0.96$ (je nach Gewichtung und Regularisierung)
		\item $D_f \approx 2.94$ in vielen Fits
	\end{itemize}
