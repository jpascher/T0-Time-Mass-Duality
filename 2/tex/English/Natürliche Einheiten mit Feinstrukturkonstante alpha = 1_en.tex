	\begin{equation}
		\lambda_0 = \frac{c}{2\pi \nu_0}

	\begin{equation}
		\lambda_0 = \frac{h}{m_0 c}

		h = 2\pi m_0 c \lambda_0 = \frac{2\pi m_0 c^2}{\nu_0} = \frac{2\pi E_0}{\nu_0}
