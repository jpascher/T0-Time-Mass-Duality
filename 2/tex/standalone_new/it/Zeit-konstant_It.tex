\chapter{Constantes Temporelles}
\section{Introduction}
Les échelles de temps fondamentales dans la théorie T0.
\section{Temps de Planck}
\begin{equation}
t_P = \sqrt{\frac{\hbar G}{c^5}} \approx 5,39 \times 10^{-44} \, \text{s}
\end{equation}
\section{Temps caractéristique $\xi$}
\begin{equation}
\tau_\xi = \frac{L_\xi}{c} \approx 3,3 \times 10^{-13} \, \text{s}
\end{equation}
\begin{insight}[title=Hiérarchie temporelle]
Les échelles de temps cosmiques émergent de $\tau_\xi$ par des puissances de $\xi$.
\end{insight}
