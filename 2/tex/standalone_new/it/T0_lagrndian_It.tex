\chapter{Le Lagrangien T0}

\section{Introduction}

Le Lagrangien est la fonction centrale de la mécanique classique et de la théorie des champs.

\section{Lagrangien classique}

\begin{equation}
	L = T - V = \frac{1}{2}m\dot{x}^2 - V(x)
\end{equation}

\section{Lagrangien du Modèle Standard}

\begin{equation}
	\mathcal{L}_{SM} = \mathcal{L}_{jauge} + \mathcal{L}_{Higgs} + \mathcal{L}_{fermions} + \mathcal{L}_{Yukawa}
\end{equation}

\begin{insight}[title=Extension T0]
	Le Lagrangien T0 ajoute un terme $\xi$ :
	\begin{equation}
		\mathcal{L}_{T0} = \mathcal{L}_{SM} + \mathcal{L}_\xi
	\end{equation}
\end{insight}

\section{Terme $\xi$}

\begin{equation}
	\mathcal{L}_\xi = -\frac{1}{2}\partial_\mu \phi_\xi \partial^\mu \phi_\xi - V(\phi_\xi)
\end{equation}

où $\phi_\xi$ est le champ $\xi$.
