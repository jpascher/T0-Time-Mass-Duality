\chapter{Constante Gravitationnelle}

\section{Introduction}

La constante gravitationnelle $G$ est l'une des constantes fondamentales les moins bien mesurées. La théorie T0 offre une dérivation de $G$ à partir du paramètre $\xi$.

\section{Valeur expérimentale}

\begin{equation}
	G = 6,674 \times 10^{-11} \, \text{m}^3 \, \text{kg}^{-1} \, \text{s}^{-2}
\end{equation}

\section{Dérivation T0}

Dans la théorie T0, $G$ est liée à $\xi$ par :

\begin{equation}
	G = \frac{\hbar c}{m_P^2} = f(\xi)
\end{equation}

\begin{insight}[title=Relation fondamentale]
	La constante gravitationnelle émerge de la géométrie du champ $\xi$ :
	\begin{equation}
		G \cdot \xi = \text{constante dimensionnelle}
	\end{equation}
\end{insight}

\section{Masse de Planck}

La masse de Planck :

\begin{equation}
	m_P = \sqrt{\frac{\hbar c}{G}} \approx 2,18 \times 10^{-8} \, \text{kg}
\end{equation}

Dans la théorie T0, cette masse est liée au paramètre $\xi$ par :

\begin{equation}
	m_P = \xi^{-n} \cdot m_e
\end{equation}

\section{Implications}

\begin{enumerate}
	\item Unification de la gravitation et de la mécanique quantique
	\item Explication de la faiblesse de la gravitation
	\item Prédiction de nouveaux effets gravitationnels
\end{enumerate}
