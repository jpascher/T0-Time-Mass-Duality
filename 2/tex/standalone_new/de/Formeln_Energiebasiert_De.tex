\chapter{Energiebasierte Formeln der T0-Theorie}

\begin{abstract}
	Die T0-Theorie verwendet ein energiebasiertes Framework, in dem alle physikalischen Größen durch Energie und Energieverhältnisse ausgedrückt werden. Dieses Dokument sammelt die wichtigsten energiebasierten Formeln.
\end{abstract}

\section{Fundamentale Beziehungen}

\subsection{Die Zeit-Energie-Dualität}

Die Basis aller Formeln ist:
\begin{equation}
	\boxed{T \cdot E = 1}
\end{equation}

\subsection{Masse-Energie}

Die Masse-Energie-Beziehung:
\begin{equation}
	E = mc^2 \quad \Leftrightarrow \quad m = \frac{E}{c^2}
\end{equation}

In natürlichen Einheiten ($c = 1$): $E = m$.

\section{Teilchenmassen}

\subsection{Allgemeine Massenformel}

Alle Teilchenmassen folgen aus:
\begin{equation}
	m_i = \frac{K_{\text{frak}}}{\xi_i}
\end{equation}

wobei $\xi_i = \xi_0^{n_i}$ für Teilchen $i$.

\section{Anomale magnetische Momente}

Die T0-Formel für anomale magnetische Momente:
\begin{equation}
	\Delta a_\ell^{\text{T0}} = \frac{5\xi^4}{96\pi^2\lambda^2} \cdot m_\ell^2
\end{equation}

\section{Kosmologie}

\subsection{Hubble-Parameter}

\begin{equation}
	H = H_0 \cdot f(\xi, z)
\end{equation}

\subsection{Kosmologische Konstante}

\begin{equation}
	\Lambda_{\text{eff}} = \frac{\xi^2}{r_0^2}
\end{equation}

\begin{keyresult}
	\textbf{Zusammenfassung der T0-Formeln}
	
	Alle Formeln enthalten den Parameter $\xi = \frac{4}{3} \times 10^{-4}$:
	\begin{itemize}
		\item Massen: $m \sim 1/\xi^n$
		\item Anomale Momente: $\Delta a \sim \xi^4$
		\item Kosmologie: $\Lambda \sim \xi^2$
	\end{itemize}
\end{keyresult}
