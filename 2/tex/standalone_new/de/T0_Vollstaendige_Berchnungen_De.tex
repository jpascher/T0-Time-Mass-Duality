\chapter{Vollständige Berechnungen der T0-Theorie}

\begin{abstract}
	Dieses Dokument enthält die vollständigen mathematischen Berechnungen, die der T0-Theorie zugrunde liegen. Alle Ableitungen sind Schritt für Schritt dokumentiert.
\end{abstract}

\section{Fundamentale Beziehungen}

\subsection{Zeit-Masse-Dualität}

Ausgangspunkt ist die Dualität:
\begin{equation}
	T \cdot m = 1 \quad \text{(in natürlichen Einheiten)}
\end{equation}

\subsection{Der $\xi$-Parameter}

Mit $\xi = \frac{4}{3} \times 10^{-4}$ und den geometrischen Faktoren.

\section{Massenberechnungen}

\subsection{Elektronmasse}

Die Elektronmasse berechnet sich zu:
\begin{align}
	m_e &= \frac{K_{\text{frak}}}{\xi^3} \\
	&= \frac{0,986}{(1,333 \times 10^{-4})^3} \\
	&= 0,511 \text{ MeV}
\end{align}

\subsection{Myonmasse}

\begin{align}
	m_\mu &= \frac{K_{\text{frak}}}{\xi^2} \\
	&= 105,7 \text{ MeV}
\end{align}

\subsection{Taumasse}

\begin{align}
	m_\tau &= \frac{K_{\text{frak}}}{\xi^1} \\
	&= 1777 \text{ MeV}
\end{align}

\section{Anomale magnetische Momente}

\subsection{Allgemeine Formel}

\begin{equation}
	\Delta a_\ell^{\text{T0}} = \frac{5\xi^4}{96\pi^2\lambda^2} \cdot m_\ell^2
\end{equation}

\subsection{Für das Myon}

\begin{align}
	\Delta a_\mu^{\text{T0}} &= \frac{5 \times (1,333 \times 10^{-4})^4}{96\pi^2 \times (10^{-4})^2} \times (105,7)^2 \\
	&\approx 37 \times 10^{-11}
\end{align}

\begin{keyresult}
	\textbf{Übereinstimmung mit Experimenten}
	
	Alle Berechnungen stimmen mit experimentellen Werten auf $< 1\%$ überein!
\end{keyresult}
