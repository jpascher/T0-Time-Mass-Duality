Ce document présente les calculs complets et vérifications de la théorie T0.

\section{Calcul du paramètre $\xi$}

\begin{equation}
	\xi = \frac{4}{3} \times 10^{-4}
\end{equation}

\section{Vérification de la température du CMB}

\begin{equation}
	T_{CMB} = \frac{16}{9} \xi^2 \cdot E_\xi = 2,725 \, \text{K}
\end{equation}

\begin{insight}[title=Précision]
	Accord avec l'observation : > 99,9\%
\end{insight}

\end{equation}
