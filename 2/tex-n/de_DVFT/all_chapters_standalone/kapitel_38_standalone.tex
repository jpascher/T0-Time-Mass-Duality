\begin{equation}

\end{equation}

\begin{equation}

\end{equation}

\begin{equation}
	\Box \rho + \frac{dU}{d\rho} + \xi \cdot \rho \cdot \nabla^2 \mathcal{F}(r) = T^{00}.
\end{equation}

\begin{equation}
	\rho_{\max} \approx \rho_0 \cdot \xi^{-3/2},

da höhere Terme dominieren.

Krümmung bleibt endlich:
\begin{equation}
	R_{\max} \approx \frac{c^4}{G \hbar} \cdot \xi^2.
\end{equation}
