% Patch (P14-Konsolidierung): firstpage-Seitenstil auch für Buch-Builds
% Einige (Standalone-)Kapitel rufen \thispagestyle{firstpage}; im Buch-
% Preamble war dieser Stil bisher nicht definiert. Idempotent definiert.
% ------------------------------------------------------------------
\makeatletter
\@ifundefined{ps@firstpage}{%
  \fancypagestyle{firstpage}{%
    \fancyhf{}%
    \fancyfoot[C]{\thepage}%
    \renewcommand{\headrulewidth}{0pt}%
  }%
}{}
\makeatother
