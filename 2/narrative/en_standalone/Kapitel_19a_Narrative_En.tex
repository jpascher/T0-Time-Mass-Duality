	\section*{The Cosmological Constant Problem in Standard QFT}
	
	The vacuum energy density is calculated as the sum of zero-point energies of all modes:
	
	\begin{equation}
		\rho_{\text{vac}}^{\text{QFT}} = \int_0^{k_{\text{Planck}}} \frac{1}{2} \hbar \omega_k \frac{d^3k}{(2\pi)^3} \propto k_{\max}^4
	\end{equation}
	
	The integral diverges quartically with the cutoff $k_{\max}$. At the Planck scale $k_{\max} \approx 10^{35} \, \text{m}^{-1}$, this yields a theoretical density of about $10^{113} \, \text{kg/m}^3$, while observations show only $10^{-27} \, \text{kg/m}^3$—a discrepancy of 120 orders of magnitude.
	
	\section*{Fractal Correlation Structure of the Vacuum Phase}
	
	The correlation function of the phase is:
	
	\begin{equation}
		C(r) = \xi \ln \left( \frac{|r| + l_0}{l_0} \right) + \frac{\xi^2}{2} \left[ \ln \left( \frac{|r| + l_0}{l_0} \right) \right]^2 + \mathcal{O}(\xi^3)

	It arises from resummation of self-similar contributions at each hierarchy level:
	
	\begin{equation}
		C(r) = \sum_{k=0}^\infty \xi^k C_0(r \xi^{-k})
	\end{equation}
	
	This produces long-range but controlled phase correlations due to the small factor $\xi$.
	
	The mean squared phase fluctuation in a volume $V$ is:
	
	\begin{equation}
		\langle (\Delta \theta)^2 \rangle_V = \xi \ln(V / l_0^3) + \xi^{1/2} \sqrt{V / l_0^3}
	\end{equation}
	
	The logarithmic term dominates for large volumes and prevents explosive divergences.
	
	\section*{Regulated Zero-Point Energy}
	
	The energy of a mode $k$ arises from the vacuum stiffness $B = \rho_0^2 \xi^{-2}$:
	
	\begin{equation}
		E_k = \frac{1}{2} B |\nabla \theta_k|^2 V
	\end{equation}
	
	The phase gradient scales fractally:
	
	\begin{equation}
		|\nabla \theta_k| \approx k \sqrt{\xi \ln(k l_0)}
	\end{equation}
	
	Thus, the mode energy becomes:
	
	\begin{equation}
		E_k = \frac{1}{2} B k^2 \xi \ln(k l_0) V
	\end{equation}
	
	The additional $\ln(k l_0)$ damps higher modes logarithmically rather than linearly.
	
	The total vacuum energy is the integral up to the natural fractal cutoff $k_{\max} \approx \pi \xi^{-1} / l_0$:
	
	\begin{equation}
		E_{\text{total}} = \int \frac{d^3k}{(2\pi)^3} \frac{1}{2} B k^2 \xi \ln(k l_0) \, V
	\end{equation}
	
	The dominant contribution of the integral:
	
	\begin{equation}

	After division by $V$ and inserting the factors, a finite density emerges:

	Numerically, accounting for the $\rho_0$ scaling, $\rho_{\text{vac}}$ matches exactly the observed dark energy ($\Omega_\Lambda \approx 0.7$).
