\chapter*{Notation and Symbol Definitions}
\addcontentsline{toc}{chapter}{Notation and Symbol Definitions}

This central reference explains all mathematical symbols and notations used throughout this book.

\section*{Fundamental Quantities}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$T_0$] Fundamental time constant (Big Bang time)
    \item[$M_{\text{tot}}$] Total mass of the universe
    \item[$c$] Speed of light
    \item[$G$] Gravitational constant
    \item[$\hbar$] Reduced Planck constant
    \item[$k_B$] Boltzmann constant
\end{itemize}

\section*{Geometric Quantities}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$R_{\text{Uni}}$] Radius of the observable universe
    \item[$R_S$] Schwarzschild radius
    \item[$a(t)$] Scale factor
    \item[$H(t)$] Hubble parameter
    \item[$g_{\mu\nu}$] Metric tensor
    \item[$\Gamma^\lambda_{\mu\nu}$] Christoffel symbols
    \item[$R_{\mu\nu}$] Ricci tensor
    \item[$R$] Ricci scalar (curvature scalar)
\end{itemize}

\section*{Energy and Density}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$\rho$] Energy density
    \item[$\rho_c$] Critical density
    \item[$\rho_m$] Matter density
    \item[$\rho_r$] Radiation density
    \item[$\rho_\Lambda$] Dark energy density (cosmological constant)
    \item[$p$] Pressure
    \item[$w$] Equation of state parameter ($w = p/\rho$)
\end{itemize}

\section*{Cosmological Parameters}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$H_0$] Hubble constant today
    \item[$\Omega_m$] Matter density parameter
    \item[$\Omega_r$] Radiation density parameter
    \item[$\Omega_\Lambda$] Dark energy parameter
    \item[$\Omega_k$] Curvature parameter
    \item[$\Omega_{\text{tot}}$] Total density parameter
    \item[$z$] Redshift
\end{itemize}

\section*{Field-Theoretic Quantities}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$\phi$] Scalar field
    \item[$\psi$] Wave function
    \item[$A_\mu$] Vector potential
    \item[$F_{\mu\nu}$] Field strength tensor
    \item[$T_{\mu\nu}$] Energy-momentum tensor
    \item[$\mathcal{L}$] Lagrangian density
    \item[$S$] Action
\end{itemize}

\section*{Quantum Mechanical Quantities}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$\hat{H}$] Hamiltonian operator
    \item[$\hat{p}$] Momentum operator
    \item[$\hat{x}$] Position operator
    \item[$\langle \cdot \rangle$] Expectation value
    \item[$[\hat{A}, \hat{B}]$] Commutator
    \item[$\Delta x$] Uncertainty/standard deviation
\end{itemize}

\section*{Thermodynamic Quantities}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$S$] Entropy
    \item[$T$] Temperature
    \item[$U$] Internal energy
    \item[$F$] Free energy
    \item[$\beta$] Inverse temperature ($\beta = 1/(k_B T)$)
    \item[$Z$] Partition function
\end{itemize}

\section*{Special Symbols and Operators}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$\nabla$] Nabla operator (gradient)
    \item[$\nabla^2$, $\Delta$] Laplace operator
    \item[$\Box$] d'Alembertian operator
    \item[$\partial_\mu$] Partial derivative ($\partial/\partial x^\mu$)
    \item[$\nabla_\mu$] Covariant derivative
    \item[$\delta_{ij}$] Kronecker delta
    \item[$\epsilon_{\mu\nu\rho\sigma}$] Levi-Civita tensor
\end{itemize}

\section*{Units and Constants}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item[$\SI{}{\meter}$] Meter
    \item[$\SI{}{\second}$] Second
    \item[$\SI{}{\kilogram}$] Kilogram
    \item[$\SI{}{\joule}$] Joule (energy)
    \item[$\SI{}{\electronvolt}$] Electronvolt
    \item[$\SI{}{\year}$] Year
    \item[$\text{Gly}$] Gigalightyear ($10^9$ light years)
    \item[$\text{Mpc}$] Megaparsec
\end{itemize}

\section*{Mathematical Conventions}

\begin{itemize}[leftmargin=2cm, labelsep=0.5cm]
    \item Greek indices ($\mu, \nu, \rho, \sigma$): Spacetime indices $0,1,2,3$
    \item Latin indices ($i, j, k$): Spatial indices $1,2,3$
    \item Einstein summation convention: Repeated indices are summed
    \item Metric signature: $(-,+,+,+)$ (mostly-plus convention)
    \item Natural units: $c = \hbar = k_B = 1$ (unless stated otherwise)
\end{itemize}

\vspace{1cm}
\noindent
This notation guide serves as a quick reference. Detailed definitions and physical meanings are provided in the respective chapters.
