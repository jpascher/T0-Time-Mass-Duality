\section{Energie-Feld-Qubits und die T0-Darstellung}

\textbf{[B]} Standard-Qubits werden als komplexe Vektoren
$\alpha|0\rangle+\beta|1\rangle\in\mathbb{C}^2$ im Hilbertraum dargestellt. T0
verwendet stattdessen Zylinderkoordinaten $(z,r,\theta)$:

\begin{itemize}[leftmargin=2em,itemsep=2pt,topsep=2pt]
  \item $z\in[-1,1]$: Projektion auf die Berechnungsachse ($z=+1
        \Leftrightarrow|0\rangle$, $z=-1\Leftrightarrow|1\rangle$)
  \item $r\in[0,1]$: Superpositionsamplitude (radialer Abstand), $z^2+r^2=1$
  \item $\theta\in[0,2\pi)$: Phase (Azimutwinkel)
\end{itemize}

\textbf{[S]} Der entscheidende konzeptionelle Wechsel: $r^2$ ist keine
Wahrscheinlichkeit, sondern repräsentiert \emph{Energiedichte} des
Superpositionszustands. Das ermöglicht deterministische Evolution bei
Beibehaltung quantenmechanischer Interferenz. Die Verbindung zur Standardform ist
dieselbe Brücke wie in A070: $\alpha=\sqrt{(1+z)/2}\,e^{i\theta/2}$.

\textbf{[B]} Die zylindrische Darstellung ist für atomare Systeme ($R/r\gg1$) die
Näherung der vollständigen Torusgeometrie. Für Protonen-Skalensysteme ist das
Seitenverhältnis $R/r\sim2{,}5\times10^{18}$ -- die Näherung ist im Grenzfall
exakt. Numerische Vergleiche zylindrischer und toroidaler Berechnungen weichen
für $R/r>10^{12}$ um weniger als $0{,}02\,\%$ voneinander ab.

\textbf{[K]} Das modifizierte Hadamard-Gatter mit Bell-Dämpfung für ein $n$-Qubit-System:
\[
  H_{\text{T0}}^{(n)}:\; (z,r,\theta)\;\mapsto\;
  \bigl(r\cdot e^{-\xi\ln(n)/D_f},\;z\cdot e^{-\xi\ln(n)/D_f},\;
  \theta+\tfrac\pi2\bigr).
\]
Die Transformation $(z,r)\to(r,z)$ implementiert den Basiswechsel; der
Dämpfungsfaktor $\exp(-\xi\ln(n)/D_f)$ stabilisiert die Multi-Qubit-Verschränkung.

\section{Die \texorpdfstring{$\phi$}{phi}-hierarchische QFT}

\textbf{[B]} Die $\phi$-hierarchische QFT wendet Phasen $2\pi/\phi^k$ statt
$2\pi/2^k$ an, mit $\phi=(1+\sqrt5)/2$:
\[
  \phi\text{-QFT}:\;|x\rangle\;\mapsto\;
  \frac{1}{\sqrt{Q_\phi}}\sum_{y=0}^{Q_\phi-1} e^{2\pi i xy/Q_\phi}|y\rangle,
  \qquad Q_\phi=\phi^n.
\]

\textbf{[B]} \textbf{Äquivalenzsatz.} Für Shors Algorithmus zur Faktorisierung
von $N<2^{20}$ mit Fehlerwahrscheinlichkeit $\delta<10^{-6}$ gilt:
\[
  P_{\text{Erfolg}}(\text{Standard-QFT})\;\le\;
  P_{\text{Erfolg}}(\phi\text{-QFT})\;\le\;
  P_{\text{Erfolg}}(\text{Standard-QFT}) + \xi.
\]
Der Beweis verläuft in drei Schritten: (1) Für jede Periode $r\in[2,N]$
existiert $k\in\mathbb Z$ mit $|r-\phi^k\cdot c|<1/(2r^2)$ für eine rationale
Zahl $c$ mit kleinem Nenner (Weylscher Gleichverteilungssatz, Dirichlet-Schubfach).
(2) Diese Störung erfüllt die Kettenbruchstabilität
$|y/Q-s/r|<1/(2r^2)$. (3) Die Bell-Dämpfung erhöht das Signal-zu-Rausch-Verhältnis:
$\text{SNR}_{\phi\text{-QFT}} = \text{SNR}_\text{std}\cdot(1+\xi\ln(r)/D_f)$,
was $P_{\phi\text{-QFT}}\ge P_\text{std}$ sichert.

\textbf{[K]} \textbf{Dekohärenzunterdrückung.} Unter Phasenrauschen
$\epsilon\cdot\sigma_z$ mit $\epsilon\sim\mathcal N(0,\sigma^2)$ gilt für
$\epsilon<0{,}1$:
\[
  \text{Fidelity}_{\phi\text{-QFT}}
  = \text{Fidelity}_\text{std}\cdot\exp\!\bigl(\xi\epsilon^2/D_f\bigr)
  > \text{Fidelity}_\text{std}.
\]
Der Fehlerterm ist quadratisch in $\epsilon$ statt linear -- das ist der
strukturelle Stabilitätsvorteil.

\section{Bell-Test-Modifikationen und CHSH-Skalierung}

\textbf{[K]} Die T0-modifizierte Bell-Korrelation für zwei Qubits mit Messwinkeln
$a$ und $b$:
\[

\]
mit $f(n,l,j)=(n/\phi)^l\cdot(1+\xi j/\pi)$. Für photon-ähnliche Qubits
($n=1,l=0,j=1$): $f\approx1{,}000042$.

\textbf{[K]} \textbf{CHSH-Skalierungsformel.} Für $n$ verschränkte Qubits:
\[
  \boxed{\;\text{CHSH}^{\text{T0}}(n)
  = 2\sqrt2\cdot\exp\!\Bigl(-\frac{\xi\ln n}{D_f}\Bigr)\;}

$E_i^{\text{T0}}\approx E_i^{\text{QM}}\cdot(1-\xi\ln n/D_f)$; Summation über

\section{Beweis der \texorpdfstring{$\phi$}{phi}-QFT-Äquivalenz: vollständige Herleitung}

\textbf{[B]} \textbf{Lemma (Periodenabdeckung durch $\phi$-Potenzen).} Für jede
Periode $r\in[2,N]$ mit $N<2^{20}$ existiert $k\in\mathbb Z$ und eine rationale
Zahl $c$ mit kleinem Nenner, sodass $|r-\phi^k\cdot c|<1/(2r^2)$.

Beweis: Der goldene Schnitt $\phi=(1+\sqrt5)/2$ ist eine Pisot-Zahl. Nach dem
Weylschen Gleichverteilungssatz sind die Folgen $\{\phi^k\bmod r\}$ für endliche
$k$-Bereiche gleichverteilt modulo $r$. Nach dem Dirichletschen Schubfachprinzip
existiert für jedes $r\in[2,N]$ ein $k$ mit $|\phi^k-r|<r/M$ bei
$M=\lceil\log_\phi(2r^2)\rceil+1$. Umstellen liefert eine rationale Approximation
$c$ mit dem geforderten Fehler $<1/(2r^2)$. Für $N<2^{20}$ genügen $k$-Werte
bis $\lceil20/\log_2\phi\rceil\approx36$.

\textbf{[B]} \textbf{Hauptsatz.} Der Beweis verläuft in fünf Teilen:

\textbf{(A) Signalanalyse.} Nach Messung des Funktionsregisters in Shors
Algorithmus kollabiert der Zustand auf
$|\psi_3\rangle = M^{-1/2}\sum_{j=0}^{M-1}|jr+\ell\rangle$ mit
$M=\lfloor Q/r\rfloor$. Die QFT liefert Amplitude
$\alpha(y) = (QM)^{-1/2}\,e^{2\pi i\ell y/Q}\sum_{j=0}^{M-1}e^{2\pi ijry/Q}$.
Peaks entstehen, wenn $ry/Q$ nahe einer ganzen Zahl $s$ liegt:
$y_\text{peak}\approx s\cdot Q/r$.

\textbf{(B) $\phi$-QFT-Modifikation.} Ersetze $Q=2^n$ durch $Q_\phi=\phi^n$;
die Bell-Dämpfung $\mathcal D(\theta)=\exp(-\xi\theta^2/(\pi^2D_f))$ modifiziert
die Amplitude. Das unterdrückt Off-Peak-Beiträge (wo $\Delta\phi$ signifikant
von einem ganzzahligen Vielfachen von $2\pi$ abweicht) und wirkt als Filter.

\textbf{(C) Peak-Positionen.} Hauptpeaks bei
$y_{\phi,\text{peak}}\approx s\cdot Q_\phi/r$. Nach dem Lemma existiert $k$ mit
$|Q_\phi/r - 2^n/r\cdot\phi^k/2^k|<0{,}2\cdot2^n/r$.

\textbf{(D) Kettenbruchstabilität.} Der Kettenbruchalgorithmus extrahiert $r$
aus $y/Q$ unter der Bedingung $|y/Q-s/r|<1/(2r^2)$. Unser Fehler aus (C) ist
$0{,}2/r<1/(2r^2)$ für $r\ge2$ -- die Bedingung ist erfüllt.

\textbf{(E) Erfolgswahrscheinlichkeit.} Die Standardformel gibt
$P_\text{std}=4/\pi^2-1/(3r)+O(r^{-2})$. Die Bell-Verstärkung liefert
$\text{SNR}_{\phi\text{-QFT}} = \text{SNR}_\text{std}\cdot(1+\xi\ln(r)/D_f)$,
sodass $P_{\phi\text{-QFT}}\ge P_\text{std}$. Die obere Schranke
$P_\text{std}+\xi$ folgt, da $\phi$-QFT perfekten Erfolg nicht überschreiten kann.

\section{Dekohärenz-Unterdrückung: vollständiger Beweis}

\textbf{[K]} Unter Phasenrauschen $\epsilon\sim\mathcal N(0,\sigma^2)$ gilt für
Standard-QFT: $|\alpha_\text{std}(y)|\to|\alpha_\text{std}(y)|\cdot(1-|\epsilon|)+O(\epsilon^2)$
(lineare Degradation). Für $\phi$-QFT mit Bell-Dämpfung
$\mathcal D(\theta)=\exp(-\xi\theta^2/(\pi^2D_f))$:
\[
  |\alpha_\phi(y)|\;\to\;|\alpha_\phi(y)|\cdot
  \exp\!\Bigl(-\frac\xi{D_f}\epsilon^2\Bigr)
  \;\approx\;|\alpha_\phi(y)|\cdot\Bigl(1-\frac\xi{D_f}\epsilon^2\Bigr).
\]
Da $\xi\epsilon/D_f\ll1$ für $\epsilon<0{,}1$ ist $1-\xi\epsilon^2/D_f>1-\epsilon$.
Der Fehlerterm ist quadratisch in $\epsilon$ statt linear. Für $\sigma=0{,}1$:
Fehler Standard-QFT $\approx10\,\%$, Fehler $\phi$-QFT
$\approx\xi\sigma^2/D_f\approx4{,}4\times10^{-7}$.

class T0Qubit:
    def apply_bell_damping(self, n_qubits):
        d = exp(-self.XI * log(n_qubits) / self.D_F)
        self.z *= d; self.r *= d
        norm = sqrt(self.z**2 + self.r**2)
        self.z /= norm; self.r /= norm

    def apply_hadamard_t0(self, n_qubits):
        self.z, self.r = self.r, self.z   # Basiswechsel
        self.apply_bell_damping(n_qubits)
        self.theta = (self.theta + pi/2) % (2*pi)
\end{verbatim}
