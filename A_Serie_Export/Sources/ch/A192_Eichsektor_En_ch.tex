\section{Purpose}

\textbf{[STIPULATION]} A190 benennt den Eichsektor als größte offene Lücke.
Dieses Dokument führt die Ableitung aus: von der Torus-Topologie zur

\section{Starting Point: Torus and Flux Quantisation}

\textbf{[B]} Auf $T^4$ existieren nicht-kontrahierbare Schleifen. Der Fluss
eines $U(1)$-Zusammenhangs durch eine solche Schleife $C$ ist quantisiert:
jede stetige Deformation von $C$ lässt den Fluss invariant. Das ist die
Grundstruktur aller drei Eichgruppen.

\section{$U(1)_Y$: Electromagnetic Charge}

\textbf{[B]} Die Wellenfunktion $\psi$ akkumuliert beim Umlauf um die
Torus-Schleife $C$ eine Phase:
\[
  \psi \;\longrightarrow\; \psi\cdot\exp\!\Bigl(i\,\frac{e}{\hbar}\,\Phi\Bigr),
  \qquad \Phi = \oint_C A_\mu\,dx^\mu = n\,\frac{h}{e},\quad n\in\mathbb{Z}.
\]
Die Monodromie $\exp(i\cdot 2\pi n)=1$ ist für alle $n\in\mathbb{Z}$ erfüllt.
Das ist die $U(1)$-Eichstruktur: die Phase $e^{i\alpha(x)}$ mit
$\alpha(x) = (e/\hbar)\int A_\mu dx^\mu$.

\textbf{[B]} Daraus folgt die kovariante Ableitung und der Feldstärketensor:
\[
  D_\mu = \partial_\mu + ieA_\mu,
  \qquad
  F_{\mu\nu} = \partial_\mu A_\nu - \partial_\nu A_\mu,
  \qquad
  \mathcal{L}_\text{EM} = -\tfrac14\,F_{\mu\nu}F^{\mu\nu}.
\]
Der Eichlagrangian folgt aus der minimalen Kopplung: $F_{\mu\nu}$ ist die
Krümmung der Verbindung $A_\mu$, $\mathcal{L}_\text{EM}$ ist deren
Norm. Kein freier Schritt.

\section{$SU(3)_C$: Colour Charge from Linking Number}

\textbf{[B]} Drei Flussfäden auf $T^4/\mathbb{Z}_3$ tragen Verschlingungszahlen
$\ell k(i,j)\in\mathbb{Z}$ für $i,j\in\{1,2,3\}$ (Fäden $i$ und $j$).
Die unabhängigen nicht-trivialen Zustände sind:

\begin{center}
\renewcommand{\arraystretch}{1.3}
{\small\begin{tabular}{lll}
\toprule
Verschlingungszustand & Generator & Physik \\
\midrule
$\ell k(1,2)=\pm1$, Rest~0 & $\lambda_1,\lambda_2$ & Gluon $r\leftrightarrow g$ \\
$\ell k(1,3)=\pm1$, Rest~0 & $\lambda_4,\lambda_5$ & Gluon $r\leftrightarrow b$ \\
$\ell k(2,3)=\pm1$, Rest~0 & $\lambda_6,\lambda_7$ & Gluon $g\leftrightarrow b$ \\
$\ell k(1,1)-\ell k(2,2)$  & $\lambda_3$           & Gluon neutral (1) \\
$\ell k(1,1)+\ell k(2,2)-2\,\ell k(3,3)$ & $\lambda_8$ & Gluon neutral (2) \\
\bottomrule
\end{tabular}}
\renewcommand{\arraystretch}{1.0}
\end{center}

\textbf{[B]} Zählung: $\binom{3}{2}=3$ Paare, je Real- und Imaginärteil,
gibt $6$ außerdiagonale; plus $2$ spurfreie Diagonale (Rang-$1$-Reduktion
der $3\times3$-Spurfreiheitsbedingung) gibt $6+2=8$ unabhängige Generatoren
-- exakt $\dim\bigl(su(3)\bigr)$. Die Eichgruppe $SU(3)_C$ ist topologisch
erzwungen, nicht postuliert.

\textbf{[B]} Kovariante Ableitung und Lagrangian analog zu $U(1)$:
\[
  D_\mu = \partial_\mu + ig_s\,G_\mu^a\,\lambda_a/2,
  \qquad
  G_{\mu\nu}^a = \partial_\mu G_\nu^a - \partial_\nu G_\mu^a
    + g_s\,f^{abc}\,G_\mu^b\,G_\nu^c,
\]
\[
  \mathcal{L}_\text{stark} = -\tfrac14\,G_{\mu\nu}^a\,G^{a,\mu\nu}.
\]
Die Nichtkommutativität $f^{abc}\neq0$ folgt aus der Nichtkommutativität der
Verschlingungszahl-Operationen für drei Fäden.

\section{$SU(2)_L$: Weak Isospin Group from Chiral Projection}

\textbf{[B]} Die chirale Projektion $P_+$ (A095) wählt den $m\ge0$-Sektor
des Torus. Der Spinor-Dublett-Charakter der schwachen Wechselwirkung folgt
aus der Spin-$\tfrac12$-Struktur (Wicklung $(1,1)$, A263): die zwei
Komponenten des Isospin-Dubletts $(u_L, d_L)$ bzw. $(\nu_L, e_L)$ sind die
zwei unabhängigen $m=0$- und $m=+1$-Sektoren nach der Projektion.

\textbf{[B]} Die drei Generatoren $\tau^+, \tau^-, \tau^3$ (Pauli-Matrizen)
der $SU(2)_L$ entsprechen den drei unabhängigen Übergängen zwischen den
zwei projizierten Zuständen. Kovariante Ableitung:
\[
  D_\mu = \partial_\mu + ig_w\,W_\mu^a\,\sigma_a/2,
  \qquad
  W_{\mu\nu}^a = \partial_\mu W_\nu^a - \partial_\nu W_\mu^a
    + g_w\,\varepsilon^{abc}\,W_\mu^b\,W_\nu^c,
\]
\[
  \mathcal{L}_\text{schwach} = -\tfrac14\,W_{\mu\nu}^a\,W^{a,\mu\nu}.
\]

\section{Overall Picture: Gauge Lagrangian from Torus Topology}

\textbf{[B]} Der vollständige Eichlagrangian des Standardmodells:
\[
  \mathcal{L}_\text{Eich}
  = -\tfrac14\,G_{\mu\nu}^a G^{a,\mu\nu}
  - \tfrac14\,W_{\mu\nu}^a W^{a,\mu\nu}
  - \tfrac14\,F_{\mu\nu}F^{\mu\nu}
\]
folgt aus drei topologischen Strukturen des $T^4/\mathbb{Z}_3$:
\begin{itemize}[leftmargin=2em,itemsep=2pt,topsep=2pt]
  \item $U(1)_Y$: Flussquantisierung durch nicht-kontrahierbare Schleife
  \item $SU(3)_C$: Verschlingungszahl dreier $\mathbb{Z}_3$-Flussfäden
    ($6+2=8$ Generatoren)

\end{itemize}

\section{Open Questions}

\textbf{Die Kopplungskonstanten $g_s$, $g_w$, $e$ sind nicht aus $\xi$
hergeleitet.} Die Gruppenstruktur folgt aus der Topologie; die Stärke
der Kopplungen nicht. Das ist die verbleibende Lücke.

\textbf{Das Laufen der Kopplungen} ($g_s(\mu)$, $g_w(\mu)$ als Funktionen
der Energieskala $\mu$) ist nicht aus FFGFT hergeleitet.

\textbf{Der Higgs-Mechanismus} -- die spontane Symmetriebrechung
$SU(2)_L\times U(1)_Y\to U(1)_\text{EM}$ -- ist nicht aus der
Torus-Geometrie abgeleitet. Die Higgs-EFT-Konsistenzprüfung (A190,
$2{.}3\,\%$) ist ein Hinweis, kein Beweis.
